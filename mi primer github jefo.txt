// formulas para compra y venta

// Precio de venta = $2
// Precio de compra = $1

void venta(int *fresas, int *naranjas, int *mangos, int *platanos, int *guayabas, float *dinero){

    int fruta, cantidad;
    printf("Que fruta deseas vender?\n");
    printf("1. Fresas\n2. Naranjas\n3. Mangos\n4. Platanos\n5. Guayabas\n");
    printf("Opcion: ");
    scanf("%d", &fruta);

    printf("Cantidad a vender: ");
    scanf("%d", &cantidad);

    if (cantidad <= 0){
        printf("Cantidad invalida.\n");
        system("pause");
        return;
    }

    int *inv;

    switch(fruta){
        case 1: inv = fresas; break;
        case 2: inv = naranjas; break;
        case 3: inv = mangos; break;
        case 4: inv = platanos; break;
        case 5: inv = guayabas; break;
        default:
            printf("Opcion invalida.\n");
            system("pause");
            return;
    }

    if (*inv < cantidad){
        printf("No tienes suficiente inventario.\n");
        system("pause");
        return;
    }

    *inv -= cantidad;
    *dinero += cantidad * 2;

    printf("Venta realizada. Ganaste $%d\n", cantidad * 2);
    system("pause");
    system("cls");
}


void compra(int *fresas, int *naranjas, int *mangos, int *platanos, int *guayabas, float *dinero){

    int fruta, cantidad;
    printf("Que fruta deseas comprar?\n");
    printf("1. Fresas\n2. Naranjas\n3. Mangos\n4. Platanos\n5. Guayabas\n");
    printf("Opcion: ");
    scanf("%d", &fruta);

    printf("Cantidad a comprar: ");
    scanf("%d", &cantidad);

    if (cantidad <= 0){
        printf("Cantidad invÃ¡lida.\n");
        system("pause");
        return;
    }

    float costo = cantidad * 1;

    if (*dinero < costo){
        printf("No tienes suficiente dinero.\n");
        system("pause");
        return;
    }

    int *inv;

    switch(fruta){
        case 1: inv = fresas; break;
        case 2: inv = naranjas; break;
        case 3: inv = mangos; break;
        case 4: inv = platanos; break;
        case 5: inv = guayabas; break;
        default:
            printf("Opcion invalida.\n");
            system("pause");
            return;
    }

    *inv += cantidad;
    *dinero -= costo;

    printf("Compra realizada. Gastaste $%.2f\n", costo);
    system("pause");
    system("cls");
}
